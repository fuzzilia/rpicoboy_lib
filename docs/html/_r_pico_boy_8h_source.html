300void RPicoBoy::fpsDelay() {
301 unsigned int new_time = micros();
302 if (new_time > _old_micros) {
303 unsigned int diff_time = new_time - _old_micros;
304 if (_delay_time > diff_time) {
305 unsigned int wait_time = _delay_time - diff_time;
306 delayMicroseconds(wait_time);
307 }
308 } else {
309 // 計算上おかしい場合は単純に1フレーム分待つ
310 delayMicroseconds(_delay_time);

515void RPicoBoy::drawImage64bitBin(int16_t x, int16_t y, int w, int h, int s, long long unsigned int *array, long long unsigned int *mArray) {
516 int result = 0;
517
518 for (int i = 0; i < h; i++) {
519 // 16進数の1行分を2進数で取得
520 char bin_c[w];
521 for (int j = (w-1); j >=0; j--) {
522 result = (array[i] >> j) & 1;
523 bin_c[(w-1)-j] = '0'+result;
524 }
525
526 // マスクがあればそっちも2進数で取得
527 char bin_c_m[w];
528 if (mArray != NULL) {
529 for (int j = (w-1); j >=0; j--) {
530 result = (mArray[i] >> j) & 1;
531 bin_c_m[(w-1)-j] = '0'+result;
532 }
533 }
534
535 // 必要な分だけ表示
536 for (int k = 0; k < s; k++) {
537 if (mArray != NULL) {
538 // マスクがある場合
539 drawPixelWithMask(x+k, y+i, bin_c[k], bin_c_m[k]);
540 } else {
541 // マスクがない場合
542 drawPixelWithMask(x+k, y+i, bin_c[k], ' ');
543 }
544 }
545 }

856void RPicoBoy::ledOn(int r, int g, int b, int value) {
857 int r_val = (r >= LED_MAX) ? LED_MAX : r;
858 int g_val = (g >= LED_MAX) ? LED_MAX : g;
859 int b_val = (b >= LED_MAX) ? LED_MAX : b;
860 int val = (value >= LUX_MAX) ? LUX_MAX : value;
861
862 // rのみ電圧が低いので計算で調整
863 r_val = r_val * RED_VOLTAGE / PICO_VOLTAGE;
864
865 // 値が逆なのでMAXの値から引く
866 r_val = LED_MAX - r_val;
867 g_val = LED_MAX - g_val;
868 b_val = LED_MAX - b_val;
869
870 // 光量を調節して光らせる
871 ledOff();
872 analogWrite(_anode_pin, val);
873 analogWrite(_r_pin, r_val);
874 analogWrite(_g_pin, g_val);
875 analogWrite(_b_pin, b_val);
876}

1037void RPicoBoy::writeEEPROM(unsigned int eeaddress, byte data) {
1038 // 最大番地より大きい場合は何もしない
1039 if (eeaddress > MAX_EEADDRESS) { return; }
1040
1041 // 番地を指定してデータ書き込み
1042 Wire1.beginTransmission(EEPROM_ADDRESS);
1043 Wire1.write((int)(eeaddress >> 8)); // 番地のHighByte
